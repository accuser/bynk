commons d {
	fn check(res: Option[Result[Int, E]]) -> String {
		match res {
			Some(Ok(n)) if n > 0 => "pos"
			Some(Ok(_)) => "nonpos"
			Some(Err(_)) => "err"
			None => "none"
		}
	}
}
